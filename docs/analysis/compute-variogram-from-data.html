cpdef double compute_distance(int i, int j, double[:, :, :] P):
    cdef double d = 0.0
    cdef int nsnps = P.shape[1]
    cdef int l  = 0
    cdef int k1 = 0
    cdef int k2 = 0
    for l in range(nsnps):
         for k1 in [0,1,2]:
            for k2 in [0,1,2]:
                d += (k1-k2)*(k1-k2) * P[i, l, k1] * P[j, l, k2]
    return(d)
